window.agregar = async function () {
  if (!auth.currentUser) {
    alert("Debes iniciar sesión primero");
    return;
  }

  const nombre = document.getElementById("nombre").value;
  const desc = document.getElementById("desc").value;
  const precio = document.getElementById("precio").value;
  const categoria = document.getElementById("categoria").value
    .toLowerCase()
    .normalize("NFD")
    .replace(/[\u0300-\u036f]/g, "")
    .trim();
  const file = document.getElementById("file").files[0];

  const guardar = async (imgURL) => {
    await addDoc(collection(db, "productos"), {
      nombre,
      desc,
      precio,
      categoria,
      img: imgURL
    });
  };

  if (file) {
    const reader = new FileReader();
    reader.onload = async (e) => {
      await guardar(e.target.result);
    };
    reader.readAsDataURL(file);
    return;
  }

  await guardar("https://images.unsplash.com/photo-1563729784474-d77dbb933a9e?w=300");
};

// EDITAR
window.editar = async function (id, data) {

  if (!esAdmin) {
    alert("No autorizado");
    return;
  }

  const nombre = prompt("Nombre:", data.nombre);
  if (nombre === null) return;

  const desc = prompt("Descripción:", data.desc);
  if (desc === null) return;

  const precio = prompt("Precio:", data.precio);
  if (precio === null) return;

  await updateDoc(doc(db, "productos", id), {
    nombre,
    desc,
    precio
  });
};

// ELIMINAR
window.eliminar = async function (id) {

  if (!esAdmin) {
    alert("No autorizado");
    return;
  }

  if (confirm("¿Eliminar producto?")) {
    await deleteDoc(doc(db, "productos", id));
  }
};

        console.log("RAW:", p.categoria);

        const contenedor = cats[cat];

        if (!contenedor) {

window.toggleAdmin = function () {
